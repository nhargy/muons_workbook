import numpy as np
import matplotlib.pyplot as plt

def transc(h0):
    A = 68e-6
    B = 2e-6
    a = 0.285
    b = 0.698

    expression  = A * np.exp(-h0/a) + B * np.exp(-h0/b)

    return expression
